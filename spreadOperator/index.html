Spread Operator and Rest Parameter

Rest Parameter - Collects a bunch of items and puts them in an array or an object

Spread Operator - It allows an iterable to spread or expand individually

Deconstructing - Deconstructing means taking out indivisual items from an object or array and storing them in variables.

        
            // Rest Operators
            function sendCars(...ids){
                console.log("Ids---", ids);
            }

            sendCars(100, 200, 300, 400);

            // Spread Operator
            function sendBikes(bike1, bike2, bike3){
                console.log('Bikes ids are', bike1, bike2, bike3);
            }

            let allBikeIds = [11, 12, 13];

            sendBikes(...allBikeIds);

            let bike = { id: 120, model: 2019};
            let newBike = bike;
            let newBikeWithColor = {...bike, color: 'black'}
            console.log("New Bike = ",newBike);
            console.log("Added New Property in existing object",newBikeWithColor);

            // Deconstructing
            let carIds = [1,2,3];
            let [car1, car2, car3] = carIds;
            console.log("Deconstructed array elements : ", car1, car2, car3);

            let car = { id: 120, model: 2019};
            let {id, model} = car;
            console.log("Deconstructed object", id, model);

Output

        
            Ids--- (4) [100, 200, 300, 400]
            Bikes ids are 11 12 13
            New Bike =  {id: 120, model: 2019}
            Added New Property in existing object {id: 120, model: 2019, color: 'black'}
            Deconstructed array elements :  1 2 3
            Deconstructed object : 120 2019
