Functions of array

push() - It inserts an element to the end into the array

popup() - It removes the last element from the array

shift() - It removes the first element of an array and returns that element.

unshift() - It adds new element to the beginning of array and returns the new length.

forEach() - Used to iterate array elements in a simple fashion

map() - Used for creating a mapped array.
mapped array : an array created after applying some transformation to the existing array
Size of mapped array will be always same as the size of the initial array

filter() - filter the array elements based on some condition

find() - find a particular element in an array.

reverse() - It reverses the order of elements in an array. It will make change to the existing array.

concat() - It joins two or more arrays and returns joined array.

slice() - It returns a new array object copying to it all items from start to end(exclusive)
- This method selects from a given start, up to a(not inclusive) given end.
- This method does not change the original array.
Syntax - array.slice(start, end)

start Optional
Start position. Default is 0.
Negative numbers select from the end of the array.
end Optional.
End position. Default is last element.
Negative number select from the end of the array.

splice() - This method adds and removes array elements.
This method overwrites the original array.
Syntax - array.splice(index, count, item1, ....., itemX) 

        
            let fruits = ["Orange", "Apple", "Grapes", "Pineapple"]

            console.log("Original array", fruits);
        
            console.log("===== push() =====")
            fruits.push("Banana");
            console.log("Fruits after pushing Banana" , fruits);
        
            console.log("===== pop() =====")
            fruits.pop();
            console.log("fruits after poping" , fruits);
        
            console.log("===== shift() =====")
            fruits.shift();
            console.log("fruits after shift()" , fruits);
        
            console.log("===== unshift() =====")
            fruits.unshift("Guava");
            console.log("fruits after unshift()" , fruits);
        
            console.log("===== forEach() =====")
            fruits.forEach((element, key) => {
                console.log("Array elements : ", key,":",element)
            });
        
            console.log("===== map() =====")
            let mappedArray = fruits.map((fruit)=>{
                if(fruit === 'Banana' || fruit === 'Apple'){
                    return fruit.toUpperCase()
                }else{
                    return fruit;
                }
            });
            console.log("Mapped array :" , mappedArray)
        
            console.log("===== filter() =====")
            let filteredFruits = fruits.filter((fruit) =>{
                return fruit === 'Apple'
            })
            console.log("Filtered array", filteredFruits);
        
            console.log("===== find() =====")
            let foundFruit = fruits.find((fruit) => {
                return fruit === 'Apple'
            });
            console.log("Element found", foundFruit)
        
            console.log("==== reverse() ====");
            fruits.reverse();
            console.log("Reversed array", fruits);
        
            console.log("==== concat() ====");
            let vegtables = ["Patato", "Tamato", "Pumpkin"]
            let joinedArray = fruits.concat(vegtables);
            console.log("Joined array", joinedArray);
        
            console.log("==== slice() ====");
            let slicedArray = fruits.slice(1,3);
            console.log("Sliced array", fruits.slice(), fruits.slice(1), slicedArray);
        
            console.log("==== splice() ====")
            // At position 2, add "Lemon" and "Kiwi"
            fruits.splice(2, 0, "Lemon", "Kiwi");
            console.log("Spliced array : ", fruits)
        
            // At poistion 2, remove 2 items
            fruits.splice(2, 2);
            console.log("Spliced array : ", fruits);
        
            // At position 2, remove 1 item, add "Lemon" and "Kiwi"
            fruits.splice(2, 1, "Lemon", "Kiwi");
            console.log("Spliced array : ", fruits);
        
            // Remove all items from position 2. because we are not passing count parameter.
            fruits.splice(2);
            console.log("Spliced array : ", fruits);
